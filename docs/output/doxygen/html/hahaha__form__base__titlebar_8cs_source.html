158 /// <summary>
159 /// 攔截非用戶區訊息，讓表單自行處理外觀與縮放命中測試。
160 /// </summary>
161 protected override void WndProc(ref Message m)
162 {
163 switch (m.Msg)
164 {
165 // 移除預設非用戶區，讓自訂標題列可以吃滿整個視窗上緣。
166 case WM_NCCALCSIZE:
167 if (m.WParam != IntPtr.Zero) { m.Result = IntPtr.Zero; return; }
168 break;
169
170 // 阻止作業系統繪製原生邊框與標題列。
171 case WM_NCPAINT:
172 m.Result = IntPtr.Zero;
173 return;
174
175 // 避免焦點切換時自訂框線被系統重新刷掉。
176 case WM_NCACTIVATE:
177 m.Result = (IntPtr)1;
178 return;
179
180 //// 改標題文字、改 Icon 都會觸發非用戶區重繪，一併吃掉重畫
181 //case WM_SETTEXT:
182 //case WM_SETICON:
183 // base.WndProc(ref m); // 讓文字/圖示真的被更新
184 // // 然後阻止隨後的系統非用戶區重畫
185 // SuppressNonClientRepaint();
186 // return;
187
188 //// 主題切換時，避免系統回畫標題列；同時重套一次 DWM 設定
189 //case WM_THEMECHANGED:
190 // ApplyNoNonClientDrawing();
191 // m.Result = IntPtr.Zero;
192 // return;
193
194 //// 這兩個是 UXTheme 用來畫標題列/框線的「內部訊息」，直接吃掉
195 //case WM_NCUAHDRAWCAPTION:
196 //case WM_NCUAHDRAWFRAME:
197 // m.Result = IntPtr.Zero;
198 // return;
199 case WM_NCHITTEST:
200 base.WndProc(ref m);
201
202 // 由於原生非用戶區已關閉，這裡要自行定義縮放邊界命中結果。
203 Point p = PointToClient(new Point((int)m.LParam & 0xFFFF, (int)m.LParam >> 16));
204 bool top = p.Y < Form_Border_Width_;
205 bool left = p.X < Form_Border_Width_;
206 bool right = p.X > Width - Form_Border_Width_;
207 bool bottom = p.Y > Height - Form_Border_Width_;
208
209 if (top && left) { m.Result = (IntPtr)HTTOPLEFT; return; }
210 if (top && right) { m.Result = (IntPtr)HTTOPRIGHT; return; }
211 if (bottom && left) { m.Result = (IntPtr)HTBOTTOMLEFT; return; }
212 if (bottom && right) { m.Result = (IntPtr)HTBOTTOMRIGHT; return; }
213 if (top) { m.Result = (IntPtr)HTTOP; return; }
214 if (left) { m.Result = (IntPtr)HTLEFT; return; }
215 if (right) { m.Result = (IntPtr)HTRIGHT; return; }
216 if (bottom) { m.Result = (IntPtr)HTBOTTOM; return; }
217 return;
218 }
219
220 base.WndProc(ref m);
221 }
