77 private bool Parse_Minutes_Scaled_1e7(string s, ref int pos, int end, out long minutes_scaled_1e7)
78 {
79 // NMEA 分鐘欄位格式：mm[.ffff...]
80 minutes_scaled_1e7 = 0;
81
82 if (!Parse_U32_Fixed_Digits(s, ref pos, end, 2, out uint minutesInt))
83 return false;
84
85 long scaled = (long)minutesInt * 10_000_000L;
86
87 if (pos < end && s[pos] == '.')
88 {
89 pos++;
90
91 long frac = 0;
92 int fracDigits = 0;
93 while (pos < end && Is_Digit(s[pos]) && fracDigits < 7)
94 {
95 frac = (frac * 10) + (s[pos] - '0');
96 pos++;
97 fracDigits++;
98 }
99
100 // 最多保留 7 位小數，並使用第 8 位做四捨五入。
101 bool roundUp = false;
102 if (pos < end && Is_Digit(s[pos]))
103 {
104 roundUp = s[pos] >= '5';
105 while (pos < end && Is_Digit(s[pos])) pos++;
106 }
107
108 if (fracDigits < 7)
109 {
110 // 補齊到固定 1e7 比例，後續就能用整數運算。
111 long mul = 1;
112 for (int i = 0; i < (7 - fracDigits); i++) mul *= 10;
113 frac *= mul;
114 }
115
116 if (roundUp) frac++;
117
118 // 小數進位回整數分鐘。
119 if (frac >= 10_000_000L)
120 {
121 frac -= 10_000_000L;
122 scaled += 10_000_000L;
123 }
124
125 scaled += frac;
126 }
127
128 if (pos != end) return false;
129 minutes_scaled_1e7 = scaled;
130 return true;
131 }

159 private bool Parse_Rmc_Lat_Lon_E7(string nmea, out int latitude_e7, out int longitude_e7)
160 {
161 latitude_e7 = 0;
162 longitude_e7 = 0;
163
164 if (nmea == null) return false;
165 if (nmea.Length < 12) return false; // 最短合法格式類似 "$??RMC,,,,,,,"
166
167 // 只要句型是 RMC，就接受任意 talker ID。
168 if (nmea[0] != '$') return false;
169 if (nmea.Length < 7) return false;
170 if (nmea[3] != 'R' || nmea[4] != 'M' || nmea[5] != 'C' || nmea[6] != ',') return false;
171
172 // 欄位 1：UTC 時間，目前不使用。
173 int p = 7;
174 int comma = Find_Char(nmea, p, ',');
175 if (comma < 0) return false;
176
177 // 欄位 2：狀態，必須是 'A' 才視為有效定位。
178 p = comma + 1;
179 comma = Find_Char(nmea, p, ',');
180 if (comma < 0) return false;
181 if (p >= comma) return false;
182 char status = nmea[p];
183 if (status != 'A') return false;
184
185 // 欄位 3：緯度。
186 p = comma + 1;
187 comma = Find_Char(nmea, p, ',');
188 if (comma < 0 || comma == p) return false;
189 if (!Parse_Coord_Ddmm_To_E7(nmea, p, comma, 2, out int latAbsE7)) return false;
190
191 // 欄位 4：緯度半球。
192 p = comma + 1;
193 comma = Find_Char(nmea, p, ',');
194 if (comma < 0) return false;
195 if (p >= comma) return false;
196 char ns = nmea[p];
197 if (ns != 'N' && ns != 'S') return false;
198
199 // 欄位 5：經度。
200 p = comma + 1;
201 comma = Find_Char(nmea, p, ',');
202 if (comma < 0 || comma == p) return false;
203 if (!Parse_Coord_Ddmm_To_E7(nmea, p, comma, 3, out int lonAbsE7)) return false;
204
205 // 欄位 6：經度半球。
206 p = comma + 1;
207 if (p >= nmea.Length) return false;
208 char ew = nmea[p];
209 if (ew != 'E' && ew != 'W') return false;
210
211 latitude_e7 = (ns == 'S') ? -latAbsE7 : latAbsE7;
212 longitude_e7 = (ew == 'W') ? -lonAbsE7 : lonAbsE7;
213 return true;
214 }
